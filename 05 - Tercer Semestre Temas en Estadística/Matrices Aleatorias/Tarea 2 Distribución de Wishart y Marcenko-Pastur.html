import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
sns.set_style("darkgrid")

p = 30
n = [30, 300]
Matriz = []
for i in n :
    Z = np.random.normal(0, 1, (p,i))
    a1 = np.array([12,11,10,9,8,7])
    a2 = np.repeat(3, p-6)
    D2 = np.diag(np.concatenate((a1,a2)))
    D = np.sqrt(np.diag(np.concatenate((a1,a2))))
    X = np.dot(D,Z)
    S = np.dot(X, X.T)/ i
    v, w = np.linalg.eigh(S)
    temp = np.sort(v)[::-1]
    Matriz.append(temp)

final = []
p = 30
n = [30, 300]

for j in n:
    valores_p = []
    for i in range(1000):
        X = np.random.normal(0, 1, (p,j))
        temp2 = np.dot(X,X.T)/ j
        v, w = np.linalg.eigh(temp2)
        valores_p.append(v.tolist())
    flattened = [val for sublist in valores_p for val in sublist]
    final.append(flattened)

En t√©rminos matem√°ticos lo que sucede es lo siguiente:

$C_{1,2}$ viene de distribuci√≥n con $\mu=0$ y $\sigma=1/\sqrt(n)$, y conforme $n \rightarrow \infty$, $C_{1,2} \rightarrow 0$
